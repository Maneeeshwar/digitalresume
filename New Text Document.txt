    <script>
        function showScene(num) {
            // Hide all scenes
            for(let i = 1; i <= 5; i++) {
                document.getElementById('scene' + i).classList.add('hidden');
            }
            // Show target scene
            document.getElementById('scene' + num).classList.remove('hidden');
            
            // Trigger animations
            if(num === 2) {
                setTimeout(() => {
                    document.getElementById('scanLine').classList.remove('hidden');
                }, 500);
            }
        }
        
        function toggleScan() {
            const scanLine = document.getElementById('scanLine');
            scanLine.classList.toggle('hidden');
            if(!scanLine.classList.contains('hidden')) {
                setTimeout(() => {
                    scanLine.classList.add('hidden');
                }, 4000);
            }
        }
        
        function playVideo() {
            alert('Video intro would play here - 30 second professional introduction with candidate speaking confidently to camera');
        }
        
        // Initialize first scene
        showScene(1);
        
        // Text scramble effect for final scene
        const chars = '!<>-_\\/[]{}—=+*^?#________';
        const finalText = document.getElementById('finalText');
        
        function scrambleText(element, finalText) {
            let iteration = 0;
            const interval = setInterval(() => {
                element.innerText = finalText
                    .split('')
                    .map((letter, index) => {
                        if(index < iteration) {
                            return finalText[index];
                        }
                        return chars[Math.floor(Math.random() * chars.length)];
                    })
                    .join('');
                
                if(iteration >= finalText.length){ 
                    clearInterval(interval);
                }
                iteration += 1/3;
            }, 30);
        }
        
        // Trigger scramble when scene 5 is shown
        const observer = new MutationObserver((mutations) => {
            mutations.forEach((mutation) => {
                if(mutation.target.id === 'scene5' && !mutation.target.classList.contains('hidden')) {
                    scrambleText(finalText, 'GET NOTICED.');
                }
            });
        });
        
        observer.observe(document.getElementById('scene5'), { attributes: true, attributeFilter: ['class'] });
    </script>
